 22 Class({
 23 	name : "Animator",
 24 	pack : motion,
 25 	/**
 26 	* @name constructor
 27 	* @memberOf motion.Animator
 28 	* @function
 29 	* @param object объект jQuery, содержащий набор элементов для применения анимации
 30 	* @param params объект, содержащий наборы инструкций и параметров для анимации элементов, переданных параметром object
 31 	* @description
 32 	* <b>Конструктор класса</b> : <i>Да</i>
 33 	*/
 34 	constructor : function(object, params){
 35 		object.each(function(){
 36 			console.log(typeof params.property);
 37 			if (typeof params.property == "object"){
 38 				var n;
 39 				var props = params.property;
 40 				for (n in props){
 41 					var _params = new tools.copy(params);
 42 					_params.property = _params.property[n];
 43 					for (m in _params){
 44 						if (typeof _params[m] == "object" && m != 'property'){
 45 							_params[m] = (_params[m][n] != undefined) ? _params[m][n] : (_params[m].pop() || _params[m]);
 46 						}
 47 					}
 48 					console.log(_params);
 49 					temp = new motion.Instance($(this), _params);
 50 				}
 51 			}
 52 			else{
 53 				temp = new motion.Instance($(this), params);
 54 			}
 55 		});
 56 	},
 57 	public : {},
 58 	protected : {
 59 		defaultInterval : 10,
 60 		defaultDuration : 1000,
 61 		update : function (targetObject, property, val){
 62 			console.log('changing '+property+' to '+val);
 63 			targetObject.css(property, val);
 64 		},
 65 		tweeningArray : function (first, last, step){
 66 			var a = [];
 67 			var l = Math.ceil(Math.abs((last-first)/step));
 68 			for (var i = first, j = 0; j < l; i += step, j++){
 69 				a.push(i);
 70 			}
 71 			console.log('l='+l+' length='+a.length);
 72 			if (a[l] != last) {a.push(last);}
 73 			return a;
 74 		},
 75 		step : function (first, last, interval, duration){
 76 			var step = (last-first)*interval/duration;
 77 			return step;
 78 		},
 79 		getDefaultValue : funtion(object, param){
 80 			var p;
 81 			switch (param){
 82 				case 'width':
 83 					p = object.width();	break;
 84 				case 'height':
 85 					p = object.height();	break;
 86 				case 'top':
 87 					p = object.position().x;break;
 88 				case 'left':
 89 					p = object.position().y;break;
 90 				default :
 91 					p = object().css(param) || 0;break;
 92 			}
 93 			return p;
 94 		}
 95 	}

 97 /**
 98 * @name Instance
 99 * @class
100 * Класс, отвечающий за анимацию определенного свойства.
101 * @memberOf motion
102 * @description
103 * Данный класс управляет анимацией определенного свойства элемента.
104 */
105 Class({
106 	name : "Instance",
107 	pack : motion,
108 	parent : motion.Animator,
109 	/**
110 	* @name constructor
111 	* @memberOf motion.Instance
112 	* @function
113 	* @param object объект jQuery, содержащий элемент для применения анимации
114 	* @param params объект, содержащий инструкции и параметры для анимации
115 	* @description
116 	* <b>Конструктор класса</b> : <i>Да</i>
117 	*/
118 	constructor : function(object, params) {
119 		this.property = params.property;
120 		var from = (params.from == undefined) ? this.parent.getDefaultValue(object, this.property) : params.from;
121 		if (typeof from == "string"){
122 			switch (from.substring(from.length-2)){
123 				case "px":
124 					from = Number(from.substring(0, from.length-2));
125 				break;
126 				default : from = 0; break;
127 			}
128 		}
129 		var to = params.to;
130 		if(!(to == undefined)){
131 			if (typeof to == "string"){
132 				switch (to[0]){
133 					case "+": to = from + Number(to.substring(1)); break;
134 					case "-": to = from - Number(to.substring(1)); break;
135 				}
136 			}
137 			var interval = params.interval || this.parent.defaultInterval;
138 			var duration = params.duration || this.parent.defaultDuration;
139 			
140 			this.targetObject = object;
141 			this.step = this.parent.step(from, to, interval, duration);
142 			this.array = this.parent.tweeningArray(from, to, this.step);
143 			console.log(this.step);
144 			console.log(this.array);
145 			this.timer = new tools.Timer(interval, this.array.length-1);
146 			this.timer.bind('timer', this.tick);
147 			this.timer.start();
148 		}
149 	},
150 	private : {
151 		step : 0,
152 		array : [],
153 		nowAt : 1,
154 		timer : {},
155 		targetObject : {},
156 		property : "",
157 		tick : function (){
158 			this.parent.update(this.targetObject, this.property, this.array[this.nowAt]);
159 			this.nowAt++;
160 		}
161 	}
162 })
