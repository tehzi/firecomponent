 11 var tools={};
 12 /**
 13 * @name browser
 14 * @class
 15 * @memberOf tools
 16 * @example
 17 * 	Class({
 18 *		name:"some",
 19 *		implements:[tools.browser],
 20 *		constructor:function(){}
 21 *	});
 22 *	alert(new some().IE)
 23 * @description
 24 * Интерфейс, предоставляющий информацию для работы с текущей версией браузера
 25 */
 26 Class({
 27 	name:"browser",
 28 	type:"interface",
 29 	pack:tools,
 30 	/**
 31 	* @name constructor
 32 	* @memberOf tools.browser
 33 	* @description
 34 	* <b>Конструктор класса</b> : <i>Да</i>
 35 	*/
 36 	constructor:function(){},
 37 	public:{
 38 		/**
 39 		* @name browser
 40 		* @memberOf tools.browser
 41 		* @description
 42 		* <b>Область видимости</b> : <i>Публичная</i>
 43 		* <br/><b>Тип данных:</b> <i>String</i>
 44 		* <br/><b>Значение:</b> <i>Another/Firefox/Opera/Safari/Chrome/Internet Explorer</i>
 45 		* <br/>Возвращает строку с названием браузера
 46 		*/
 47 		browser:(function(){
 48 			var name="Another";
 49 			var agent=navigator.userAgent.toLowerCase();
 50 			if(agent.search("firefox")>-1){
 51 				name="Firefox";
 52 			}
 53 			if(agent.search("opera")>-1){
 54 				name="Opera";
 55 			}
 56 			if(agent.search("safari")>-1){
 57 				name="Safari";
 58 			}
 59 			if(agent.search("chrome")>-1){
 60 				name="Chrome";
 61 			}
 62 			if(agent.search("msie")>-1){
 63 				name="Internet Explorer";
 64 			}
 65 			return name;
 66 		})(),
 67 		/**
 68 		* @name IE
 69 		* @memberOf tools.browser
 70 		* @description
 71 		* <b>Область видимости</b> : <i>Публичная</i>
 72 		* <br/><b>Тип данных:</b> <i>Boolean</i>
 73 		* <br/>Проверяет является ли браузер членом семейства Internet Explorer
 74 		*/
 75 		IE:"~this.browser=='Internet Explorer'~",
 76 		/**
 77 		* @name IE6
 78 		* @memberOf tools.browser
 79 		* @description
 80 		* <b>Область видимости</b> : <i>Публичная</i>
 81 		* <br/><b>Тип данных:</b> <i>Boolean</i>
 82 		* <br/>Проверяет является ли браузер Internet Explorer'ом 6 версии
 83 		*/
 84 		IE6:"~this.IE && this.intertrigoBrowserVersion()=='6.0'~",
 85 		/**
 86 		* @name IE7
 87 		* @memberOf tools.browser
 88 		* @description
 89 		* <b>Область видимости</b> : <i>Публичная</i>
 90 		* <br/><b>Тип данных:</b> <i>Boolean</i>
 91 		* <br/>Проверяет является ли браузер Internet Explorer'ом 7 версии
 92 		*/
 93 		IE7:"~this.IE && this.intertrigoBrowserVersion()=='7.0'~",
 94 		/**
 95 		* @name IE8
 96 		* @memberOf tools.browser
 97 		* @description
 98 		* <b>Область видимости</b> : <i>Публичная</i>
 99 		* <br/><b>Тип данных:</b> <i>Boolean</i>
100 		* <br/>Проверяет является ли браузер Internet Explorer'ом 8 версии
101 		*/
102 		IE8:"~this.IE && this.intertrigoBrowserVersion()=='8.0'~",
103 		/**
104 		* @name Opera
105 		* @memberOf tools.browser
106 		* @description
107 		* <b>Область видимости</b> : <i>Публичная</i>
108 		* <br/><b>Тип данных:</b> <i>Boolean</i>
109 		* <br/>Проверяет является ли браузер Oper'ой
110 		*/
111 		Opera:"~this.browser=='Opera'~",
112 		/**
113 		* @name Firefox
114 		* @memberOf tools.browser
115 		* @description
116 		* <b>Область видимости</b> : <i>Публичная</i>
117 		* <br/><b>Тип данных:</b> <i>Boolean</i>
118 		* <br/>Проверяет является ли браузер Firefox'ом
119 		*/
120 		Firefox:"~this.browser=='Firefox'~",
121 		/**
122 		* @name Chrome
123 		* @memberOf tools.browser
124 		* @description
125 		* <b>Область видимости</b> : <i>Публичная</i>
126 		* <br/><b>Тип данных:</b> <i>Boolean</i>
127 		* <br/>Проверяет является ли браузер Chrome'ом
128 		*/
129 		Chrome:"~this.browser=='Chrome'~",
130 		/**
131 		* @name Safari
132 		* @memberOf tools.browser
133 		* @description
134 		* <b>Область видимости</b> : <i>Публичная</i>
135 		* <br/><b>Тип данных:</b> <i>Boolean</i>
136 		* <br/>Проверяет является ли браузер Safari
137 		*/
138 		Safari:"~this.browser=='Safari'~",
139 		/**
140 		* @name browserVersion
141 		* @memberOf tools.browser
142 		* @description
143 		* <b>Область видимости</b> : <i>Публичная</i>
144 		* <br/><b>Тип данных:</b> <i>String</i>
145 		* <br/>Возвращает строку c версией браузера
146 		*/
147 		browserVersion:"~this.intertrigoBrowserVersion()~"
148 	},
149 	protected:{
150 		/**
151 		* @name intertrigoBrowserVersion
152 		* @memberOf tools.browser
153 		* @type String
154 		* @description
155 		* <b>Область видимости</b> : <i>Защищенная</i>
156 		* <br/>Определяет версию браузера
157 		*/
158 		intertrigoBrowserVersion:function(){
159 			var version="None";
160 			var agent=navigator.userAgent.toLowerCase();
161 			version=agent
162 			if(this.IE){
163 				var version_detect=agent.match(/msie ([0-9]+\.[0-9]+)/i);
164 			}
165 			if(this.Opera){
166 				var version_detect=agent.match(/opera\/([0-9]+\.[0-9]+)/i);
167 			}
168 			if(this.Firefox){
169 				var version_detect=agent.match(/firefox\/([0-9]+\.[0-9]+\.[0-9]+)/i);
170 			}
171 			if(this.Chrome){
172 				var version_detect=agent.match(/chrome\/([0-9]+\.[0-9]+)/i);
173 			}
174 			if(this.Safari){
175 				var version_detect=agent.match(/version\/([0-9]+\.[0-9]+)/i);
176 			}
177 			if(version_detect){
178 				if(version_detect[1]){
179 					version=version_detect[1];
180 				}
181 			}
182 			return version;
183 		}
184 	}
