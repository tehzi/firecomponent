218 /**
219 * @name eventDispatcher
220 * @class
221 * Абстрактный класс для добавление в другие классы функций предоставляющих создание и обработку пользовательских событий 
222 * @memberOf tools
223 * @description
224 */
225 Class({
226 	pack:tools,
227 	name:"eventDispatcher",
228 	/**
229 	* @name constructor
230 	* @memberOf tools.eventDispatcher
231 	* @description
232 	* <b>Конструктор класса</b> : <i>Да</i>
233 	* <br/>Заполняет внутренний объект содержащий информацию о событиях в классе
234 	*/
235 	constructor:function(){
236 		this.eventAll={};
237 		for(var i=0;i<this.eventList.length;i++){
238 			this.eventAll[this.eventList[i]]=[];
239 		}
240 	},
241 	public:{
242 		/**
243 		* @name bind
244 		* @function
245 		* @memberOf tools.eventDispatcher
246 		* @param _event {String} Имя события
247 		* @param _function {Mixed} Ссылка на фунцию
248 		* @description
249 		* <b>Область видимости</b> : <i>Публичная</i>
250 		* <br/><b>Тип данных:</b> <i>Void</i>
251 		* <br/>Добавляет обработчик события
252 		*/
253 		bind:function(_event,_function){
254 			if(typeof _event=="string" && typeof _function=="function"){
255 				if(this.eventAll[_event]){
256 					this.eventAll[_event].push(_function);
257 					$(this).bind(_event,_function);
258 				}
259 			}
260 		},
261 		/**
262 		* @name unbind
263 		* @function
264 		* @memberOf tools.eventDispatcher
265 		* @param _event {String} Имя события
266 		* @param _function {Mixed} Ссылка на фунцию
267 		* @description
268 		* <b>Область видимости</b> : <i>Публичная</i>
269 		* <br/><b>Тип данных:</b> <i>Void</i>
270 		* <br/> Удаляет обработчик события
271 		*/
272 		unbind:function(_event,_function){
273 			if(typeof _event=="string" && typeof _function=="function"){
274 				if(this.eventAll[_event]){
275 					this.eventAll[_event]=$.grep(this.eventAll[_event],function(key,i){
276  						return key!==_function;
277 					});
278 					$(this).unbind(_event,_function);
279 				}
280 			}
281 		}
282 	},
283 	protected:{
284 		/**
285 		* @name eventList
286 		* @memberOf tools.eventDispatcher
287 		* @description
288 		* <b>Область видимости</b> : <i>Защищенная</i>
289 		* <br/><b>Тип данных:</b> <i>Array</i>
290 		* <br/>Список событий объекта
291 		*/
292 		eventList:[],
293 		/**
294 		* @name eventAll
295 		* @memberOf tools.eventDispatcher
296 		* @description
297 		* <b>Область видимости</b> : <i>Защищенная</i>
298 		* <br/><b>Тип данных:</b> <i>Object</i>
299 		* <br/>Хранилище для событий
300 		*/
301 		eventAll:{},
302 		/**
303 		* @name dispatch
304 		* @function
305 		* @memberOf tools.eventDispatcher
306 		* @param {String} _event Название события
307 		* @param {Array} _arr Список аргументов функции
308 		* @description
309 		* <b>Область видимости</b> : <i>Защищенная</i>
310 		* <br/><b>Тип данных:</b> <i>Void</i>
311 		* <br/>Выбрасывает новое событие в поток событий
312 		*/
313 		dispatch:function(_event,_arr){
314 			var __arr=[];
315 			if(typeof _arr=="object"){
316 				__arr=_arr;
317 			}
318 			if(typeof _event=="string"){
319 				if(this.eventAll[_event]){
320 					$(this).trigger(_event,__arr);
321 				}
322 			}
323 		}
324 	}

333 * <br/><b>Событие</b> : <i>timer</i>
334 * <br/>Срабатывает каждые this.delay миллисекунд
335 * <br/><b>Событие</b> : <i>timerComplete</i>
336 * <br/>Срабатывает когда таймер пройдет количество циклов указанной в переменной repeatCount
337 * @example
338 * // Таймер повторится 5 раз с интервало в 2 секунды
339 * var timer=new tools.Timer(2000,5);
340 * // Каждые 2 секунды
341 * timer.bind("timer",function(){
342 * 	//do somethink

345 * timer.bind("timerComplete",function(){
346 * 	//do somethink
347 * });

349 */

351 	pack:tools,
352 	parent:tools.eventDispatcher,
353 	name:"Timer",
354 	/**
355 	* @name constructor
356 	* @memberOf tools.Timer
357 	* @function
358 	* @param delay Задержка в миллисекундах между событиями таймера.
359 	* @param repeatCount Общее число запусков, на которое настроен таймер.
360 	* @description
361 	* <b>Конструктор класса</b> : <i>Да</i>

365 		if(typeof delay=="number"){
366 			this.delay=delay;

402 		repeatCount:0,

404 		* @name running
405 		* @memberOf tools.Timer
406 		* @default false

409 		* <br/><b>Тип данных:</b> <i>Boolean</i>

412 		running:false,
413 		/**
414 		* @name start
415 		* @memberOf tools.Timer
416 		* @function
417 		* @description
418 		* <b>Область видимости</b> : <i>Публичная</i>

420 		* <br/>Запускает таймер
421 		*/
422 		start:function(){
423 			if(this.delay>0){
424 				this.currentCount=0;
425 				this.running=true;
426 				this.timeLink=setInterval(this.timeExec,this.delay);
427 			}
428 		},
429 		/**
430 		* @name halt
431 		* @memberOf tools.Timer
432 		* @function
433 		* @description
434 		* <b>Область видимости</b> : <i>Публичная</i>
435 		* <br/>Останавливает таймер

437 		* <br/><i>Примечание: свойтсво stop недоступно в ie</i>
438 		*/
439 		halt:function(){
440 			this.running=false;
441 			clearInterval(this.timeLink);
442 		}
443 	},
444 	protected:{
445 		/**
446 		* @name eventList
447 		* @memberOf tools.Timer
448 		* @description
449 		* <b>Область видимости</b> : <i>Защищенная</i>
450 		* <br/><b>Тип данных:</b> <i>Array</i>

454 	},
455 	private:{

470 		* <br/><b>Тип данных:</b> <i>Void</i>
471 		* <br/>Вызывается каждые delay милисекунд
472 		*/
473 		timeExec:function(){
474 			this.dispatch("timer",[++this.currentCount]);
475 			if(this.repeatCount>0 && this.currentCount==this.repeatCount){
476 				this.dispatch("timerComplete",[this.currentCount]);
477 				this.halt();
478 			}
479 		}
480 	}

811 		* @function
812 		* @description
813 		* <b>Область видимости</b> : <i>Приватная</i>
814 		* <br/><b>Тип данных:</b> <i>String</i>
815 		* <br/>Вычисляет время согласно внутреннем селекторам.
816 		*/
817 		parseTime:function(time){
818 			if(typeof time=="string"){
819 				var options=time.match(/\+[0-9]+(?:m|h|d|w|M|Y)/g);
820 				var m=60000;
821 				var h=m*60;
822 				var d=h*24;
823 				var w=d*7;
824 				var todaySimp=new Date();
825 				var today={
826 					day:todaySimp.getDate(),
827 					month:todaySimp.getMonth()+1,
828 					year:todaySimp.getFullYear()
829 				};
830 				var endTime=0;
831 				for(var i=0;i<options.length;i++){
832 					var optArr=options[i].match(/\+([0-9]+)(m|h|d|w|M|Y)/);
833 					if(optArr[1] && optArr[2]){
834 						var currentTimeInt=parseInt(optArr[1]);
835 						switch(optArr[2]){
836 							case "m":{
837 								endTime+=currentTimeInt*m;
838 							}
839 							break;
840 							case "h":{
841 								endTime+=currentTimeInt*h;
842 							}
843 							break;
844 							case "d":{
845 								endTime+=currentTimeInt*d;
846 							}
847 							break;
848 							case "w":{
849 								endTime+=currentTimeInt*w;
850 							}
851 							break;
852 							case "M":{
853 								for(var j=0;j<currentTimeInt;j++){
854 									var deltaDate=new Date(today.year,today.month+j);
855 									var maxDay=Date.prototype.lastMonthDay(deltaDate.getMonth()+1,deltaDate.getFullYear());
856 									endTime+=d*maxDay;
857 								}
858 							}
859 							break;
860 							case "Y":{
861 								for(var j=1;j<=currentTimeInt;j++){
862 									endTime+=d*Date.prototype.lastYearDay(today.year+j);
863 								}
864 							}
865 							break;
866 							default:
867 						}
868 					}
869 				}
870 				var Time=new Date(new Date().getTime()+endTime);
871 				return Time.toGMTString();
872 			}
