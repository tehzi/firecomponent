 60 				name="Chrome";

140 		* @name browserVersion
141 		* @memberOf tools.browser
142 		* @description
143 		* <b>Область видимости</b> : <i>Публичная</i>
144 		* <br/><b>Тип данных:</b> <i>String</i>

151 		* @name intertrigoBrowserVersion
152 		* @function
153 		* @memberOf tools.browser
154 		* @type String
155 		* @description
156 		* <b>Область видимости</b> : <i>Защищенная</i>
157 		* <br/>Определяет версию браузера
158 		*/
159 		intertrigoBrowserVersion:function(){
160 			var version="None";
161 			var agent=navigator.userAgent.toLowerCase();
162 			version=agent
163 			if(this.IE){
164 				var version_detect=agent.match(/msie ([0-9]+\.[0-9]+)/i);
165 			}
166 			if(this.Opera){
167 				var version_detect=agent.match(/opera\/([0-9]+\.[0-9]+)/i);
168 			}
169 			if(this.Firefox){
170 				var version_detect=agent.match(/firefox\/([0-9]+\.[0-9]+\.[0-9]+)/i);
171 			}
172 			if(this.Chrome){
173 				var version_detect=agent.match(/chrome\/([0-9]+\.[0-9]+)/i);
174 			}
175 			if(this.Safari){
176 				var version_detect=agent.match(/version\/([0-9]+\.[0-9]+)/i);
177 			}
178 			if(version_detect){
179 				if(version_detect[1]){
180 					version=version_detect[1];
181 				}
182 			}
183 			return version;
184 		}

187 /**
188 * @name eventDispatcher
189 * @class
190 * Абстрактный класс для добавление в другие классы функций предоставляющих создание и обработку пользовательских событий 
191 * @memberOf tools
192 * @description
193 */
194 Class({
195 	pack:tools,
196 	name:"eventDispatcher",
197 	/**
198 	* @name constructor
199 	* @memberOf tools.eventDispatcher
200 	* @description
201 	* <b>Конструктор класса</b> : <i>Да</i>
202 	* <br/>Заполняет внутренний объект содержащий информацию о событиях в классе
203 	*/
204 	constructor:function(){
205 		for(var i=0;i<this.eventList.length;i++){
206 			this.eventAll[this.eventList[i]]=[];
207 		}
208 	},
209 	public:{
210 		/**
211 		* @name bind
212 		* @function
213 		* @memberOf tools.eventDispatcher
214 		* @param _event {String} Имя события
215 		* @param _function {Mixed} Ссылка на фунцию
216 		* @description
217 		* <b>Область видимости</b> : <i>Публичная</i>
218 		* <br/><b>Тип данных:</b> <i>Void</i>
219 		* <br/>Добавляет обработчик события
220 		*/
221 		bind:function(_event,_function){
222 			if(typeof _event=="string" && typeof _function=="function"){
223 				if(this.eventAll[_event]){
224 					this.eventAll[_event].push(_function);
225 					$(this).bind(_event,_function);
226 				}
227 			}
228 		},
229 		/**
230 		* @name unbind
231 		* @function
232 		* @memberOf tools.eventDispatcher
233 		* @param _event {String} Имя события
234 		* @param _function {Mixed} Ссылка на фунцию
235 		* @description
236 		* <b>Область видимости</b> : <i>Публичная</i>
237 		* <br/><b>Тип данных:</b> <i>Void</i>
238 		* <br/> Удаляет обработчик события
239 		*/
240 		unbind:function(_event,_function){
241 			if(typeof _event=="string" && typeof _function=="function"){
242 				if(this.eventAll[_event]){
243 					this.eventAll[_event]=$.grep(this.eventAll[_event],function(key,i){
244  						return key!==_function;
245 					});
246 					$(this).unbind(_event,_function);
247 				}
248 			}
249 		}
250 	},
251 	protected:{
252 		/**
253 		* @name eventList
254 		* @memberOf tools.eventDispatcher
255 		* @description
256 		* <b>Область видимости</b> : <i>Защищенная</i>
257 		* <br/>Список событий объекта
258 		*/
259 		eventList:[],
260 		/**
261 		* @name eventAll
262 		* @memberOf tools.eventDispatcher
263 		* @description
264 		* <b>Область видимости</b> : <i>Защищенная</i>
265 		* <br/>Хранилище для событий
266 		*/
267 		eventAll:{},
268 		/**
269 		* @name dispatch
270 		* @function
271 		* @memberOf tools.eventDispatcher
272 		* @param {String} _event Название события
273 		* @param {Array} _arr Список аргументов функции
274 		* @description
275 		* <b>Область видимости</b> : <i>Защищенная</i>
276 		* <br/>Выбрасывает новое событие в поток событий
277 		*/
278 		dispatch:function(_event,_arr){
279 			var __arr=[];
280 			if(typeof _arr=="object"){
281 				__arr=_arr;
282 			}
283 			if(typeof _event=="string"){
284 				if(this.eventAll[_event]){
285 					$(this).trigger(_event,__arr);
286 				}
287 			}
288 		}
289 	}

291 /**
292 * @name Timer
293 * @class
294 * Простой таймер облегчающий работу с периодичными функциями
295 * @memberOf tools
296 * @description
297 * <br/><b>Наследует</b> : <i>tools.eventDispatcher</i>
298 * <br/><b>Событие</b> : <i>timer</i>
299 * <br/>Срабатывает каждые this.delay миллисекунд
300 * <br/><b>Событие</b> : <i>timerComplete</i>
301 * <br/>Срабатывает когда таймер пройдет количество циклов указанной в переменной repeatCount
302 * @example
303 * // Таймер повторится 5 раз с интервало в 2 секунды
304 * var timer=new tools.Timer(2000,5);
305 * // Каждые 2 секунды
306 * timer.bind("timer",function(){
307 * 	//do somethink
308 * });
309 * // В конце 5-го цикла
310 * timer.bind("timerComplete",function(){
311 * 	//do somethink
312 * });
313 * timer.start();
314 */
315 Class({
316 	pack:tools,
317 	parent:tools.eventDispatcher,
318 	name:"Timer",
319 	/**
320 	* @name constructor
321 	* @memberOf tools.Timer
322 	* @function
323 	* @param delay Задержка в миллисекундах между событиями таймера.
324 	* @param repeatCount Общее число запусков, на которое настроен таймер.
325 	* @description
326 	* <b>Конструктор класса</b> : <i>Да</i>
327 	*/
328 	constructor:function(delay,repeatCount){
329 		this.Super();
330 		if(typeof delay=="number"){
331 			this.delay=delay;
332 		}
333 		if(typeof repeatCount=="number"){
334 			this.repeatCount=repeatCount;
335 		}
336 	},
337 	public:{
338 		/**
339 		* @name currentCount
340 		* @memberOf tools.Timer
341 		* @default 0
342 		* @description
343 		* <b>Область видимости</b> : <i>Публичная</i>
344 		* <br/>Общее число срабатываний таймера с момента его запуска с нуля.
345 		*/
346 		currentCount:0,
347 		/**
348 		* @name delay
349 		* @memberOf tools.Timer
350 		* @default 0
351 		* @description
352 		* <b>Область видимости</b> : <i>Публичная</i>
353 		* <br/>Задержка в миллисекундах между событиями таймера.
354 		*/
355 		delay:0,
356 		/**
357 		* @name repeatCount
358 		* @memberOf tools.Timer
359 		* @default 0
360 		* @description
361 		* <b>Область видимости</b> : <i>Публичная</i>
362 		* <br/>Общее число запусков, на которое настроен таймер.
363 		*/
364 		repeatCount:0,
365 		/**
366 		* @name running
367 		* @memberOf tools.Timer
368 		* @default false
369 		* @description
370 		* <b>Область видимости</b> : <i>Публичная</i>
371 		* <br/>Текущее состояние таймера: если таймер выполняется, то true, в противном случае false.
372 		*/
373 		running:false,
374 		/**
375 		* @name start
376 		* @memberOf tools.Timer
377 		* @function
378 		* @description
379 		* <b>Область видимости</b> : <i>Публичная</i>
380 		* <br/>Запускает таймер
381 		*/
382 		start:function(){
383 			if(this.delay>0){
384 				this.currentCount=0;
385 				this.running=true;
386 				this.timeLink=setInterval(this.timeExec,this.delay);
387 			}
388 		},
389 		/**
390 		* @name halt
391 		* @memberOf tools.Timer
392 		* @function
393 		* @description
394 		* <b>Область видимости</b> : <i>Публичная</i>
395 		* <br/>Останавливает таймер
396 		* <br/><i>Примечание: свойтсво stop недоступно в ie</i>
397 		*/
398 		halt:function(){
399 			this.running=false;
400 			clearInterval(this.timeLink);
401 		}
402 	},
403 	protected:{
404 		/**
405 		* @name eventList
406 		* @memberOf tools.Timer
407 		* @description
408 		* <b>Область видимости</b> : <i>Защищенная</i>
409 		* <br/>Список событий таймера
410 		*/
411 		eventList:["timer","timerComplete"]
412 	},
413 	private:{
414 		/**
415 		* @name eventList
416 		* @memberOf tools.Timer
417 		* @description
418 		* <b>Область видимости</b> : <i>Приватная</i>
419 		* <br/>Номер Интервала
420 		*/
421 		timeLink:null,
422 		/**
423 		* @name timeExec
424 		* @function
425 		* @memberOf tools.eventDispatcher
426 		* <b>Область видимости</b> : <i>Приватная</i>
427 		* <br/>Вызывается каждые delay милисекунд
428 		*/
429 		timeExec:function(){
430 			this.dispatch("timer",[++this.currentCount]);
431 			if(this.repeatCount>0 && this.currentCount==this.repeatCount){
432 				this.dispatch("timerComplete",[this.currentCount]);
433 				this.halt();
434 			}
435 		}
436 	}
