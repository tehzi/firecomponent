200 /**
201 * @name eventDispatcher
202 * @class
203 * Абстрактный класс для добавление в другие классы функций предоставляющих создание и обработку пользовательских событий 
204 * @memberOf tools
205 * @description
206 */
207 Class({
208 	pack:tools,
209 	name:"eventDispatcher",
210 	/**
211 	* @name constructor
212 	* @memberOf tools.eventDispatcher
213 	* @description
214 	* <b>Конструктор класса</b> : <i>Да</i>
215 	* <br/>Заполняет внутренний объект содержащий информацию о событиях в классе
216 	*/
217 	constructor:function(){
218 		this.eventAll={};
219 		for(var i=0;i<this.eventList.length;i++){
220 			this.eventAll[this.eventList[i]]=[];
221 		}
222 	},
223 	public:{
224 		/**
225 		* @name bind
226 		* @function
227 		* @memberOf tools.eventDispatcher
228 		* @param _event {String} Имя события
229 		* @param _function {Mixed} Ссылка на фунцию
230 		* @description
231 		* <b>Область видимости</b> : <i>Публичная</i>
232 		* <br/><b>Тип данных:</b> <i>Void</i>
233 		* <br/>Добавляет обработчик события
234 		*/
235 		bind:function(_event,_function){
236 			if(typeof _event=="string" && typeof _function=="function"){
237 				if(this.eventAll[_event]){
238 					this.eventAll[_event].push(_function);
239 					$(this).bind(_event,_function);
240 				}
241 			}
242 		},
243 		/**
244 		* @name unbind
245 		* @function
246 		* @memberOf tools.eventDispatcher
247 		* @param _event {String} Имя события
248 		* @param _function {Mixed} Ссылка на фунцию
249 		* @description
250 		* <b>Область видимости</b> : <i>Публичная</i>
251 		* <br/><b>Тип данных:</b> <i>Void</i>
252 		* <br/> Удаляет обработчик события
253 		*/
254 		unbind:function(_event,_function){
255 			if(typeof _event=="string" && typeof _function=="function"){
256 				if(this.eventAll[_event]){
257 					this.eventAll[_event]=$.grep(this.eventAll[_event],function(key,i){
258  						return key!==_function;
259 					});
260 					$(this).unbind(_event,_function);
261 				}
262 			}
263 		}
264 	},
265 	protected:{
266 		/**
267 		* @name eventList
268 		* @memberOf tools.eventDispatcher
269 		* @description
270 		* <b>Область видимости</b> : <i>Защищенная</i>
271 		* <br/>Список событий объекта
272 		*/
273 		eventList:[],
274 		/**
275 		* @name eventAll
276 		* @memberOf tools.eventDispatcher
277 		* @description
278 		* <b>Область видимости</b> : <i>Защищенная</i>
279 		* <br/>Хранилище для событий
280 		*/
281 		eventAll:{},
282 		/**
283 		* @name dispatch
284 		* @function
285 		* @memberOf tools.eventDispatcher
286 		* @param {String} _event Название события
287 		* @param {Array} _arr Список аргументов функции
288 		* @description
289 		* <b>Область видимости</b> : <i>Защищенная</i>
290 		* <br/>Выбрасывает новое событие в поток событий
291 		*/
292 		dispatch:function(_event,_arr){
293 			var __arr=[];
294 			if(typeof _arr=="object"){
295 				__arr=_arr;
296 			}
297 			if(typeof _event=="string"){
298 				if(this.eventAll[_event]){
299 					$(this).trigger(_event,__arr);
300 				}
301 			}
302 		}
303 	}

305 /**

315 * <br/>Срабатывает когда таймер пройдет количество циклов указанной в переменной repeatCount
316 * @example
317 * // Таймер повторится 5 раз с интервало в 2 секунды
318 * var timer=new tools.Timer(2000,5);
319 * // Каждые 2 секунды
320 * timer.bind("timer",function(){
321 * 	//do somethink

325 * 	//do somethink
326 * });

328 */
329 Class({
330 	pack:tools,
331 	parent:tools.eventDispatcher,
332 	name:"Timer",
333 	/**
334 	* @name constructor
335 	* @memberOf tools.Timer
336 	* @function
337 	* @param delay Задержка в миллисекундах между событиями таймера.
338 	* @param repeatCount Общее число запусков, на которое настроен таймер.
339 	* @description
340 	* <b>Конструктор класса</b> : <i>Да</i>
341 	*/
342 	constructor:function(delay,repeatCount){
343 		this.Super();
344 		if(typeof delay=="number"){
345 			this.delay=delay;
346 		}
347 		if(typeof repeatCount=="number"){
348 			this.repeatCount=repeatCount;
349 		}
350 	},
351 	public:{
352 		/**
353 		* @name currentCount
354 		* @memberOf tools.Timer

357 		* <b>Область видимости</b> : <i>Публичная</i>

363 		* @memberOf tools.Timer

367 		* <br/>Задержка в миллисекундах между событиями таймера.
368 		*/
369 		delay:0,
370 		/**
371 		* @name repeatCount
372 		* @memberOf tools.Timer

374 		* @description

377 		*/
378 		repeatCount:0,
379 		/**
380 		* @name running
381 		* @memberOf tools.Timer
382 		* @default false
383 		* @description
384 		* <b>Область видимости</b> : <i>Публичная</i>
385 		* <br/>Текущее состояние таймера: если таймер выполняется, то true, в противном случае false.
386 		*/
387 		running:false,
388 		/**
389 		* @name start
390 		* @memberOf tools.Timer
391 		* @function
392 		* @description
393 		* <b>Область видимости</b> : <i>Публичная</i>
394 		* <br/>Запускает таймер
395 		*/
396 		start:function(){
397 			if(this.delay>0){
398 				this.currentCount=0;
399 				this.running=true;
400 				this.timeLink=setInterval(this.timeExec,this.delay);
401 			}
402 		},
403 		/**
404 		* @name halt
405 		* @memberOf tools.Timer
406 		* @function
407 		* @description
408 		* <b>Область видимости</b> : <i>Публичная</i>
409 		* <br/>Останавливает таймер
410 		* <br/><i>Примечание: свойтсво stop недоступно в ie</i>
411 		*/
412 		halt:function(){
413 			this.running=false;
414 			clearInterval(this.timeLink);
415 		}
416 	},
417 	protected:{
418 		/**
419 		* @name eventList
420 		* @memberOf tools.Timer
421 		* @description
422 		* <b>Область видимости</b> : <i>Защищенная</i>
423 		* <br/>Список событий таймера
424 		*/
425 		eventList:["timer","timerComplete"]
426 	},
427 	private:{
428 		/**
429 		* @name timeLink
430 		* @memberOf tools.Timer
431 		* @description
432 		* <b>Область видимости</b> : <i>Приватная</i>
433 		* <br/>Номер Интервала
434 		*/
435 		timeLink:null,
436 		/**
437 		* @name timeExec
438 		* @function
439 		* @memberOf tools.eventDispatcher
440 		* <b>Область видимости</b> : <i>Приватная</i>
441 		* <br/>Вызывается каждые delay милисекунд
442 		*/
443 		timeExec:function(){
444 			this.dispatch("timer",[++this.currentCount]);
445 			if(this.repeatCount>0 && this.currentCount==this.repeatCount){
446 				this.dispatch("timerComplete",[this.currentCount]);
447 				this.halt();
448 			}
449 		}
450 	}

700 				var date=val.date ? this.parseTime(val.date) : "";
701 				var path=val.path ? val.path : this.defaultPath;
702 				var domain=val.domain ? val.domain : false;
703 				var secure=secure ? true : false;
704 				if(typeof name=="string" && (typeof val.val=="string" || typeof val.val=="number") && this.cookieEnabled){
705 					document.cookie=

710 						((!!secure) ? "; secure" : "")

755 				}

764 		/**
765 		* @name parseTime
766 		* @memberOf tools.CookieManager
767 		* @function
768 		* @description
769 		* <b>Область видимости</b> : <i>Приватная</i>
770 		* <br/>Вычисляет время согласно внутреннем селекторам.
771 		*/
772 		parseTime:function(time){
773 			if(typeof time=="string"){
774 				var options=time.match(/\+[0-9]+(?:m|h|d|w|M|Y)/g);
775 				var m=60000;
776 				var h=m*60;
777 				var d=h*24;
778 				var w=d*7;
779 				var todaySimp=new Date();
780 				var today={
781 					day:todaySimp.getDate(),
782 					month:todaySimp.getMonth()+1,
783 					year:todaySimp.getFullYear()
784 				};
785 				var endTime=0;
786 				for(var i=0;i<options.length;i++){
787 					var optArr=options[i].match(/\+([0-9]+)(m|h|d|w|M|Y)/);
788 					if(optArr[1] && optArr[2]){
789 						var currentTimeInt=parseInt(optArr[1]);
790 						switch(optArr[2]){
791 							case "m":{
792 								endTime+=currentTimeInt*m;
793 							}
794 							break;
795 							case "h":{
796 								endTime+=currentTimeInt*h;
797 							}
798 							break;
799 							case "d":{
800 								endTime+=currentTimeInt*d;
801 							}
802 							break;
803 							case "w":{
804 								endTime+=currentTimeInt*w;
805 							}
806 							break;
807 							case "M":{
808 								for(var j=0;j<currentTimeInt;j++){
809 									var deltaDate=new Date(today.year,today.month+j);
810 									var maxDay=Date.prototype.lastMonthDay(deltaDate.getMonth()+1,deltaDate.getFullYear());
811 									endTime+=d*maxDay;
812 								}
813 							}
814 							break;
815 							case "Y":{
816 								for(var j=1;j<=currentTimeInt;j++){
817 									endTime+=d*Date.prototype.lastYearDay(today.year+j);
818 								}
819 							}
820 							break;
821 							default:
822 						}
823 					}
824 				}
825 				var Time=new Date(new Date().getTime()+endTime);
826 				return Time.toGMTString();
827 			}
